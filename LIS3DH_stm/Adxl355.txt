#include <Arduino.h>
#include <SPI.h>
#include <math.h>

#define CS_PIN    5
#define SPI_FREQ  1000000

#define REG_DEVID_AD    0x00
#define REG_DEVID_MST   0x01
#define REG_PARTID      0x02
#define REG_STATUS      0x04
#define REG_TEMP2       0x06
#define REG_XDATA3      0x08
#define REG_FILTER      0x28
#define REG_RANGE       0x2C
#define REG_POWER_CTL   0x2D
#define REG_RESET       0x2F

#define SCALE_FACTOR_2G (1.0f / 256000.0f)

SPISettings adxl355Settings(SPI_FREQ, MSBFIRST, SPI_MODE0);

void csLow()  { digitalWrite(CS_PIN, LOW);  }
void csHigh() { digitalWrite(CS_PIN, HIGH); }

uint8_t readRegister(uint8_t reg) {
    SPI.beginTransaction(adxl355Settings);
    csLow();
    SPI.transfer((reg << 1) | 0x01);
    uint8_t result = SPI.transfer(0x00);
    csHigh();
    SPI.endTransaction();
    return result;
}

void writeRegister(uint8_t reg, uint8_t value) {
    SPI.beginTransaction(adxl355Settings);
    csLow();
    SPI.transfer((reg << 1) & 0xFE);
    SPI.transfer(value);
    csHigh();
    SPI.endTransaction();
}

void readRegisters(uint8_t startReg, uint8_t* buffer, uint8_t length) {
    SPI.beginTransaction(adxl355Settings);
    csLow();
    SPI.transfer((startReg << 1) | 0x01);
    for (uint8_t i = 0; i < length; i++) buffer[i] = SPI.transfer(0x00);
    csHigh();
    SPI.endTransaction();
}

int32_t twosComplement20(uint32_t raw) {
    raw &= 0x000FFFFF;
    if (raw & 0x00080000) return (int32_t)(raw | 0xFFF00000);
    return (int32_t)raw;
}

struct AccelData { float x, y, z; };

AccelData readAcceleration() {
    uint8_t buf[9];
    readRegisters(REG_XDATA3, buf, 9);
    AccelData data;
    data.x = twosComplement20(((uint32_t)buf[0] << 12) | ((uint32_t)buf[1] << 4) | (buf[2] >> 4)) * SCALE_FACTOR_2G;
    data.y = twosComplement20(((uint32_t)buf[3] << 12) | ((uint32_t)buf[4] << 4) | (buf[5] >> 4)) * SCALE_FACTOR_2G;
    data.z = twosComplement20(((uint32_t)buf[6] << 12) | ((uint32_t)buf[7] << 4) | (buf[8] >> 4)) * SCALE_FACTOR_2G;
    return data;
}

float readTemperature() {
    uint8_t t[2];
    readRegisters(REG_TEMP2, t, 2);
    uint16_t raw = ((uint16_t)(t[0] & 0x0F) << 8) | t[1];
    return 25.0f + ((float)raw - 1885.0f) / (-9.05f);
}

void setup() {
    Serial.begin(115200);
    delay(2000);
    Serial.println("Starting...");

    pinMode(CS_PIN, OUTPUT);
    csHigh();
    delay(100);

    SPI.begin(18, 19, 23, 5);
    Serial.println("SPI initialized");

    writeRegister(REG_RESET, 0x52);
    delay(100);
    Serial.println("Reset done");

    uint8_t devId  = readRegister(REG_DEVID_AD);
    uint8_t mstId  = readRegister(REG_DEVID_MST);
    uint8_t partId = readRegister(REG_PARTID);
    Serial.printf("DEVID_AD  : 0x%02X (expect 0xAD)\n", devId);
    Serial.printf("DEVID_MST : 0x%02X (expect 0x1D)\n", mstId);
    Serial.printf("PART ID   : 0x%02X (expect 0xED)\n", partId);

    if (devId != 0xAD || mstId != 0x1D || partId != 0xED) {
        Serial.println("ERROR: ADXL355 not found!");
        while (true) delay(1000);
    }

    writeRegister(REG_RANGE,     0x01);  // ±2g
    writeRegister(REG_FILTER,    0x33);  // ODR 500Hz, LPF 125Hz, HPF 0.08Hz (removes gravity)
    writeRegister(REG_POWER_CTL, 0x00);  // measurement mode
    delay(100);

    Serial.println("Setup complete, entering loop...");
}

void loop() {
    if (readRegister(REG_STATUS) & 0x01) {  // DATA_RDY
        AccelData a = readAcceleration();
        float total = sqrtf(a.x*a.x + a.y*a.y + a.z*a.z);
        Serial.printf(">Ax:%f\n", a.x);
        Serial.printf(">Ay:%f\n", a.y);
        Serial.printf(">Az:%f\n", a.z);
        Serial.printf(">Atotal:%f\n", total);

       
    }
    delay(2);  // 2ms matches 500 Hz ODR
}
