#include <Arduino.h>
#include <SPI.h>
#include <math.h>

// ════════════════════════════════════════════════════════════════
//  PIN & TIMING CONFIGURATION
// ════════════════════════════════════════════════════════════════
#define ADXL355_CS    4
#define SPI_CLK      18
#define SPI_MISO     19
#define SPI_MOSI     23

#define SPI_FREQ     8000000            // 8MHz SPI speed
#define SAMPLE_RATE  500                // 500 Hz Sampling
#define INTERVAL_US  (1000000UL / SAMPLE_RATE) 
#define SCALE_2G     (2.0f / 524288.0f) // 20-bit scaling for ±2g

// ADXL355 Register Map
#define REG_XDATA3    0x08
#define REG_FILTER     0x28
#define REG_RANGE      0x2C
#define REG_POWER_CTL  0x2D
#define REG_RESET      0x2F

// ════════════════════════════════════════════════════════════════
//  ANTI-ALIASING FILTER (Butterworth 2nd Order, 200Hz Cutoff)
// ════════════════════════════════════════════════════════════════
struct BiquadFilter {
    float x1=0, x2=0, y1=0, y2=0;
    // Coefficients calculated for Fc=200Hz @ Fs=500Hz
    const float b0=0.6389f, b1=1.2779f, b2=0.6389f, a1=1.1430f, a2=0.4128f;
    
    float process(float x) {
        float y = b0*x + b1*x1 + b2*x2 - a1*y1 - a2*y2;
        x2 = x1; x1 = x; y2 = y1; y1 = y;
        return y;
    }
};

BiquadFilter fX, fY, fZ;
SPISettings adxlSettings(SPI_FREQ, MSBFIRST, SPI_MODE0);

// ════════════════════════════════════════════════════════════════
//  SPI LOW-LEVEL ACCESS
// ════════════════════════════════════════════════════════════════
void adxlWrite(uint8_t reg, uint8_t val) {
    SPI.beginTransaction(adxlSettings);
    digitalWrite(ADXL355_CS, LOW);
    SPI.transfer((reg << 1) | 0x00); // Write bit = 0
    SPI.transfer(val);
    digitalWrite(ADXL355_CS, HIGH);
    SPI.endTransaction();
}

void readAccel(float &x, float &y, float &z) {
    uint8_t buf[9];
    SPI.beginTransaction(adxlSettings);
    digitalWrite(ADXL355_CS, LOW);
    SPI.transfer((REG_XDATA3 << 1) | 0x01); // Read bit = 1
    for(int i=0; i<9; i++) buf[i] = SPI.transfer(0x00);
    digitalWrite(ADXL355_CS, HIGH);
    SPI.endTransaction();

    auto parse20 = [](uint8_t h, uint8_t m, uint8_t l) {
        uint32_t raw = ((uint32_t)h << 12) | ((uint32_t)m << 4) | (l >> 4);
        if (raw & 0x80000) return (int32_t)(raw | 0xFFF00000); // Sign extend
        return (int32_t)raw;
    };

    x = parse20(buf[0], buf[1], buf[2]) * SCALE_2G;
    y = parse20(buf[3], buf[4], buf[5]) * SCALE_2G;
    z = parse20(buf[6], buf[7], buf[8]) * SCALE_2G;
}

// ════════════════════════════════════════════════════════════════
//  EXECUTION
// ════════════════════════════════════════════════════════════════
void setup() {
    Serial.begin(921600); // High baud rate is essential for 500Hz
    pinMode(ADXL355_CS, OUTPUT);
    digitalWrite(ADXL355_CS, HIGH);
    
    SPI.begin(SPI_CLK, SPI_MISO, SPI_MOSI);
    delay(100);

    adxlWrite(REG_RESET, 0x52); // Soft reset
    delay(100);

    // 0x03 = ODR 1000Hz. Higher internal ODR reduces latency.
    adxlWrite(REG_FILTER, 0x03); 
    adxlWrite(REG_RANGE, 0x01);     // ±2g
    adxlWrite(REG_POWER_CTL, 0x00); // Start measurement mode
    
    Serial.println("Timestamp_us,AccX_g,AccY_g,AccZ_g,Total_g");
}

void loop() {
    static uint32_t lastMicros = 0;
    uint32_t now = micros();

    // Strict timing loop
    if (now - lastMicros >= INTERVAL_US) {
        lastMicros += INTERVAL_US;

        float rx, ry, rz;
        readAccel(rx, ry, rz);

        // Filter signals to preserve 0-200Hz and suppress noise
        float fx = fX.process(rx);
        float fy = fY.process(ry);
        float fz = fZ.process(rz);

        // Resultant vector magnitude
        float total = sqrtf(fx*fx + fy*fy + fz*fz);

        // CSV Output
        Serial.printf(">AccX:%.6f\n>AccY:%.6f\n>AccZ:%.6f\n>Total:%.6f\n", 
                      fx, fy, fz, total);
        
    }
}

#define SAMPLE_RATE_HZ 500
#define INTERVAL_US  (1000000UL / SAMPLE_RATE_HZ) // 2000µs
#define ADXL_SCALE_2G (2.0f / 524288.0f)          // 20-bit scaling

// ADXL355 Registers
#define REG_XDATA3    0x08
#define REG_FILTER     0x28
#define REG_RANGE      0x2C
#define REG_POWER_CTL  0x2D
#define REG_RESET      0x2F

// ════════════════════════════════════════════════════════════════
//  ANTI-ALIASING FILTER (2nd Order Butterworth, 200Hz Cutoff)
// ════════════════════════════════════════════════════════════════
struct Biquad {
    float x1=0, x2=0, y1=0, y2=0;
    // Coefficients for 200Hz cutoff @ 500Hz sampling
    const float b0=0.6389f, b1=1.2779f, b2=0.6389f, a1=1.1430f, a2=0.4128f;

        x2=x1; x1=x; y2=y1; y1=y;
        return y;
    }
};

Biquad filterX, filterY, filterZ;
SPISettings settings(SPI_FREQ, MSBFIRST, SPI_MODE0);

// ════════════════════════════════════════════════════════════════
//  SPI FUNCTIONS
// ════════════════════════════════════════════════════════════════
void writeReg(uint8_t reg, uint8_t val) {
    SPI.beginTransaction(settings);
    digitalWrite(ADXL355_CS, LOW);
    SPI.transfer((reg << 1) | 0x00);
    SPI.transfer(val);
    digitalWrite(ADXL355_CS, HIGH);
    SPI.endTransaction();
}

void readAccel(float &x, float &y, float &z) {
    uint8_t buf[9];
    SPI.beginTransaction(settings);
    digitalWrite(ADXL355_CS, LOW);
    SPI.transfer((REG_XDATA3 << 1) | 0x01);
    for(int i=0; i<9; i++) buf[i] = SPI.transfer(0x00);
    digitalWrite(ADXL355_CS, HIGH);
    SPI.endTransaction();

    auto parse20 = [](uint8_t h, uint8_t m, uint8_t l) {
        uint32_t raw = ((uint32_t)h << 12) | ((uint32_t)m << 4) | (l >> 4);
        if (raw & 0x80000) return (int32_t)(raw | 0xFFF00000);

    x = parse20(buf[0], buf[1], buf[2]) * ADXL_SCALE_2G;

}

// ════════════════════════════════════════════════════════════════
//  MAIN
// ════════════════════════════════════════════════════════════════
void setup() {
    Serial.begin(921600); // High baud rate to prevent buffer lag
    pinMode(ADXL355_CS, OUTPUT);
    digitalWrite(ADXL355_CS, HIGH);
    
    SPI.begin(SPI_CLK, SPI_MISO, SPI_MOSI);
    delay(100);

    writeReg(REG_RESET, 0x52);
    delay(100);

    // ODR set to 1000Hz (0x03) for internal oversampling
    writeReg(REG_FILTER, 0x03); 
    writeReg(REG_RANGE, 0x01);     // ±2g range
    writeReg(REG_POWER_CTL, 0x00); // Measurement mode
    
    Serial.println("Timestamp_us,AccX_g,AccY_g,AccZ_g");
}

void loop() {
    static uint32_t lastMicros = 0;
    uint32_t now = micros();

    if (now - lastMicros >= INTERVAL_US) {
        lastMicros += INTERVAL_US;

        float rx, ry, rz;
        readAccel(rx, ry, rz);

        // Filter data to ensure accuracy up to 200Hz
        float fx = filterX.process(rx);
        float fy = filterY.process(ry);
        float fz = filterZ.process(rz);
        float totalAcc = sqrtf(fx*fx + fy*fy + fz*fz);
        // CSV Output
        Serial.printf("%u,%.6f,%.6f,%.6f,%.6f\n", 
                      lastMicros, fx, fy, fz, totalAcc);
    }
    }
